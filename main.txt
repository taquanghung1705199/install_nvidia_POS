##install kernel-uek 5.4.17-2136.301.1.3.el7uek.x86_64
#execute
yum update kernel-uek
yum install kernel-uek-devel
yum install gcc gcc-c++

Edit /etc/default/grub. Append the following  to “GRUB_CMDLINE_LINUX”
rd.driver.blacklist=nouveau nouveau.modeset=0

grub2-mkconfig -o /boot/efi/EFI/redhat/grub.cfg

##install r8168-8.049.02 if lost network

https://community.hetzner.com/tutorials/installing-the-r8168-driver

#execute
cd r8168-*

##change dns after install network-driver

##install driver-nvidia
#doc
https://www.advancedclustering.com/act_kb/installing-nvidia-drivers-rhel-centos-7/

#execute
sudo yum -y install epel-release
sudo yum -y install dkms

Edit/create /etc/modprobe.d/blacklist.conf and append:
blacklist nouveau

systemctl isolate multi-user.target
sh NVIDIA-Linux-x86_64-*.run


##install nvidia-docker
#doc
https://gist.github.com/tomlankhorst/33da3c4b9edbde5c83fc1244f010815c


##install driver-nvidia for docker
#execute
curl -s -L https://nvidia.github.io/nvidia-docker/rhel7.6/nvidia-docker.repo | sudo tee /etc/yum.repos.d/nvidia-docker.repo
sudo yum install --disableexcludes=all -y nvidia-container-toolkit
sudo yum install nvidia-container-runtime
sudo systemctl restart docker
docker run --gpus all nvidia/cuda:11.3.0-base-ubuntu20.04 nvidia-smi

open /etc/docker/daemon.json add:

{
  "runtimes": {
    "nvidia": {
      "path": "/usr/bin/nvidia-container-runtime",
      "runtimeArgs": []
    }
  },
  "default-runtime": "nvidia"
}

Uncommenting the following in /etc/nvidia-container-runtime/config.toml
swarm-resource = "DOCKER_RESOURCE_GPU"

sudo systemctl restart docker
